let MSI_knnfill as function(k as integer = 3, qcut as double = 0.85) -> <generic> MSIKnnFillOption {
    #
    # .NET API information
    #
    # module: MSImaging.Rscript

    #
    return call R#.interop_[Rscript::KnnFill](...);
}

Arguments

  • k: -
  • qcut: the query block area percentage threshold value,
    the higher cutoff of this parameter, the less fitting
    will be perfermen on the pixels, the lower cutoff of
    this parameter, the more interpolation will be.
